              <ref name="property"/>
            </zeroOrMore>
          </interleave>
        </element>
      </define>
  
    </div>

    Nest element - collection AND resource, with properties.

      <define name="nest">

          <interleave>
            <ref name="collection"/>
            <ref name="resource"/>

              <ref name="property"/>
            </zeroOrMore>
          </interleave>
        </element>
      </define>
