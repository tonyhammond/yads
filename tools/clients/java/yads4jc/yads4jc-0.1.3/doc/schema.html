yads: DObject Model Schema

This is the RDFS Schema for the DObject Model. Note that this is a work in progress and is open for comment. 

<?xml version="1.0" encoding="UTF-8"?>

<!--

  Resource:	doi:1014/10.system/schema.doi-schema-2001-09-03
  Comment:	RDF Schema for the DObject Model
  Author:	Tony Hammond (tony_hammond@harcourt.com)
  Date:		January 12, 2001

  Revision:	March 23, 2001 - Tony Hammond
  Revision:	July 26, 2001 - Tony Hammond, Eamonn Neylon
  Revision:	September 3, 2001 - Tony Hammond, Eamonn Neylon

-->

<RDF
  xmlns="http://www.w3.org/1999/02/22-rdf-syntax-ns#"	
  xmlns:rdf="http://www.w3.org/1999/02/22-rdf-syntax-ns#"	
  xmlns:rdfs=" http://www.w3.org/TR/2000/CR-rdf-schema-20000327#"
>

<!-- DOI Base Class -->

<rdfs:Class ID="DObject">
  <rdfs:label>DObject</rdfs:label>
  <rdfs:comment>DObject - Base class for any DObject within the DOI System.</rdfs:comment>
</rdfs:Class>

<!-- DOI Core Classes -->

<rdfs:Class ID="Agent">
  <rdfs:label>Agent</rdfs:label>
  <rdfs:comment>Agent - Class for DOI Registration Agents.</rdfs:comment>
  <rdfs:subClassOf resource="#DObject"/>
</rdfs:Class>

<rdfs:Class ID="Null">
  <rdfs:label>Null</rdfs:label>
  <rdfs:comment>Null - Class for DOI Null Resource.</rdfs:comment>
  <rdfs:subClassOf resource="#DObject"/>
</rdfs:Class>

<rdfs:Class ID="Profile">
  <rdfs:label>Profile</rdfs:label>
  <rdfs:comment>Profile - Class for DOI Application Profiles.</rdfs:comment>
  <rdfs:subClassOf resource="#DObject"/>
</rdfs:Class>

<rdfs:Class ID="Registry">
  <rdfs:label>Registry</rdfs:label>
  <rdfs:comment>Registry - Class for DOI Registries.</rdfs:comment>
  <rdfs:subClassOf resource="#DObject"/>
</rdfs:Class>

<rdfs:Class ID="Service">
  <rdfs:label>Service</rdfs:label>
  <rdfs:comment>Service - Class for DOI Services.</rdfs:comment>
  <rdfs:subClassOf resource="#DObject"/>
</rdfs:Class>

<rdfs:Class ID="Schema">
  <rdfs:label>Schema</rdfs:label>
  <rdfs:comment>Schema - Class for DOI Schemas.</rdfs:comment>
  <rdfs:subClassOf resource="#DObject"/>
</rdfs:Class>

<rdfs:Class ID="Type">
  <rdfs:label>Type</rdfs:label>
  <rdfs:comment>Type - Class for DOI Core Types (aka Properties).</rdfs:comment>
  <rdfs:subClassOf resource="#DObject"/>
</rdfs:Class>

<!-- DOI Core General Properties -->

<Property ID="access">
  <rdfs:label>access</rdfs:label>
  <rdfs:comment>access - Property attributing an access point in the DObject resource hierarchy 
                         (a string literal).</rdfs:comment>
  <rdfs:range resource=" http://www.w3.org/TR/2000/CR-rdf-schema-20000327#Literal"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="contains">
  <rdfs:label>contains</rdfs:label>
  <rdfs:comment>contains - Property attributing a container for DObject resources
                           (an RDF Container).</rdfs:comment>
  <rdfs:range resource="http://www.w3.org/TR/2000/CR-rdf-schema-20000327#Container"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="detail">
  <rdfs:label>detail</rdfs:label>
  <rdfs:comment>detail - Property attributing a DOI description element
                         (a string literal).</rdfs:comment>
  <rdfs:range resource=" http://www.w3.org/TR/2000/CR-rdf-schema-20000327#Literal"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="label">
  <rdfs:label>label</rdfs:label>
  <rdfs:comment>label - Property attributing a DOI label element
                        (a string literal).</rdfs:comment>
  <rdfs:range resource=" http://www.w3.org/TR/2000/CR-rdf-schema-20000327#Literal"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="resolve">
  <rdfs:label>resolve</rdfs:label>
  <rdfs:comment>resolve - Property attributing a DOI resolution directive
                          (a string literal).</rdfs:comment>
  <rdfs:range resource=" http://www.w3.org/TR/2000/CR-rdf-schema-20000327#Literal"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="resource">
  <rdfs:label>resource</rdfs:label>
  <rdfs:comment>resource - Property attributing a DOI resource element
                           (a generic URI).</rdfs:comment>
  <rdfs:range resource="http://www.w3.org/TR/2000/CR-rdf-schema-20000327#Resource"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="role">
  <rdfs:label>role</rdfs:label>
  <rdfs:comment>role - Property attributing a role to a DOI property
                       (a string literal).</rdfs:comment>
  <rdfs:range resource=" http://www.w3.org/TR/2000/CR-rdf-schema-20000327#Literal"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="type">
  <rdfs:label>type</rdfs:label>
  <rdfs:comment>type - Property attributing a DOI extension type
                       (a string literal).</rdfs:comment>
  <rdfs:range resource=" http://www.w3.org/TR/2000/CR-rdf-schema-20000327#Literal"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<!-- DOI Core System Properties -->

<Property ID="agent">
  <rdfs:label>agent</rdfs:label>
  <rdfs:comment>agent - Property attributing a DOI Registration Agent
                        (an Agent class DObject).</rdfs:comment>
  <rdfs:range resource="#Agent"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="null">
  <rdfs:label>null</rdfs:label>
  <rdfs:comment>null - Property attributing a DOI Null Resource
                       (a Null class DObject).</rdfs:comment>
  <rdfs:range resource="#Null"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="profile">
  <rdfs:label>profile</rdfs:label>
  <rdfs:comment>profile - Property attributing a DOI Application Profile
                          (a Profile class DObject).</rdfs:comment>
  <rdfs:range resource="#Profile"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="schema">
  <rdfs:label>schema</rdfs:label>
  <rdfs:comment>schema - Property attributing a DOI Application Schema
                         (a Schema class DObject).</rdfs:comment>
  <rdfs:range resource="#Schema"/>
  <rdfs:domain resource="#DObject"/>
</Property>

<Property ID="service">
  <rdfs:label>service</rdfs:label>
  <rdfs:comment>service -  Property attributing a DOI Service
                           (a Service class DObject).</rdfs:comment>
  <rdfs:range resource="#Service"/>
  <rdfs:domain resource="#DObject"/>
</Property>

</RDF>
